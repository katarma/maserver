# Maserver - Dockerfile (scaffold, 개발자 배포판)

FROM python:3.10-slim

# 작업 디렉토리 생성 및 이동
WORKDIR /maserver

# 필수 패키지 설치 (빌드 툴 등)
RUN apt-get update && apt-get install -y \
    build-essential \
    curl \
    && rm -rf /var/lib/apt/lists/*

# 파이썬 의존성 복사 및 설치
COPY requirements.txt .
RUN pip install --upgrade pip && pip install -r requirements.txt

# 소스 복사 (폴더/파일 scaffold)
COPY . .

# 환경변수/포트 (필요 시 추가)
# ENV PYTHONUNBUFFERED=1

# 진입점 (main.py)
CMD ["python", "main.py"]

# 개발 편의 설정 (선택적)
# EXPOSE 8000
